DELETE FROM DATOS_CREDENCIAL;

INSERT INTO DATOS_CREDENCIAL(CODIGO_PERSONAL, NOMBRES, APELLIDOS, CARRERA, COTEJADO)
SELECT 
     DISTINCT(AI.MATRICULA), 
	 DP.DATO_NOMBRES, 
	 DP.DATO_APELLIDO_PATERNO||' '||DP.DATO_APELLIDO_MATERNO,
	 UPPER(ALUMNO_EAP(AI.MATRICULA)),
	 DP.COTEJA_NOMBRE
FROM ACAD_INSCRITOS AI, DATOS_PERSONALES DP
WHERE DP.CODIGO_PERSONAL = AI.MATRICULA
ORDER BY AI.MATRICULA
